procedure principal is
 function fac(n: in out integer) return integer is
  x: integer;
 begin
  if n = 0 or n=1 then
   return 1;
  end if;
  x := fac(n-1);
  x := n*x;
  return x;
 end fac;
 
 procedure leer is
  x,y : integer;
  begin
   put("Introduzca un numero: ");
   get(x);
   y := fac(x);
   put("El resultado es: ");
   put(y);
 end leer;
begin
 leer();
end principal;
